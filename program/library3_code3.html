bool GreatNumber::operator ==(const GreatNumber &g)const{
	int psize=p.size();
	int gsize=g.p.size();

	if(psize!=gsize)return false;

	for(int i=psize-1;i>=0;i--){
		if(p[i]!=g.p[i])return false;
	}

	return true;
}
